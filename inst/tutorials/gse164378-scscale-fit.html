Gaussian Spike Model Fit With GSE164378 3’ CITE-seq

This tutorial checks the Gaussian spike model fit on the RNA side of the package GSE164378 3’ CITE-seq example. The example keeps 4,000 matched PBMC cells, 2,000 RNA HVGs, and all 228 ADT features.

Fit Spike Model

fit <- scscale_fit(
  counts,
  count_transform = count_transform,
  fit_umi = FALSE,
  mp_max_iter = 300,
  mp_grid_n = 4000
)

data.frame(
  p = fit$p,
  n = fit$n,
  c_X = fit$c_X,
  method = fit$bulk$method,
  n_spikes = sum(fit$spikes$is_spike)
)
##      p    n c_X            method n_spikes
## 1 2000 4000 0.5 median_mp_mad_gap       56

Eigenvalue Fit

lambda <- fit$spectrum$lambda
bulk <- fit$bulk
bulk_lambda <- lambda[bulk$bulk_index]
bulk_lambda <- bulk_lambda[is.finite(bulk_lambda) & bulk_lambda > 0]
spike_lambda <- lambda[bulk$spike_index]
spike_lambda <- spike_lambda[is.finite(spike_lambda) & spike_lambda > 0]
mp_density <- getFromNamespace("scscale_mp_density", "scScale")
mp_quantile <- getFromNamespace("scscale_mp_quantile_grid", "scScale")

mp_x <- seq(bulk$mp_lower, bulk$lambda_plus, length.out = 600)
mp_y <- mp_density(mp_x, c_X = fit$c_X, tau2 = bulk$tau2)

old_par <- par(mfrow = c(2, 2), mar = c(4.2, 4.2, 2.6, 1))
plot(
  seq_along(lambda),
  lambda,
  log = "y",
  pch = 16,
  cex = 0.35,
  col = ifelse(seq_along(lambda) %in% bulk$spike_index, "#D55E00", "#0072B2"),
  xlab = "Eigenvalue rank",
  ylab = expression(lambda),
  main = paste0("Log spectrum, K=", length(bulk$spike_index))
)
abline(v = length(bulk$spike_index) + 0.5, col = "grey35", lty = 2)
abline(h = bulk$lambda_plus, col = "#D55E00", lty = 2)

emp_q <- sort(bulk_lambda)
theory_q <- mp_quantile(c_X = fit$c_X, prob = ppoints(length(emp_q)), tau2 = bulk$tau2, grid_n = 4000)
plot(
  theory_q,
  emp_q,
  pch = 16,
  cex = 0.35,
  col = "#0072B2",
  xlab = "MP theoretical quantile",
  ylab = "Empirical bulk quantile",
  main = "Bulk QQ"
)
abline(0, 1, col = "#D55E00", lwd = 2)

hist(
  bulk_lambda,
  breaks = 60,
  freq = FALSE,
  col = "grey85",
  border = "white",
  xlab = expression(lambda),
  main = "Bulk eigenvalues"
)
lines(mp_x, mp_y, col = "#D55E00", lwd = 2)
abline(v = bulk$lambda_plus, col = "#D55E00", lty = 2)

hist(
  log10(bulk_lambda),
  breaks = 60,
  freq = FALSE,
  col = "grey85",
  border = "white",
  xlab = expression(log[10](lambda)),
  main = "Log bulk eigenvalues"
)
abline(v = log10(bulk$lambda_plus), col = "#D55E00", lty = 2)
rug(log10(spike_lambda), col = "#D55E00", lwd = 1)
## Warning in rug(log10(spike_lambda), col = "#D55E00", lwd = 1): some values will
## be clipped

par(old_par)
