Sequencing-Depth Scaling With GSE164378 3’ CITE-seq RNA

This tutorial tests the sequencing-depth scaling model on the RNA side of the package GSE164378 3’ CITE-seq example. ADT counts are kept in the example data for paired-modality MI tutorials, but this page only checks RNA spike scaling.

Fit UMI Scaling

The fit uses full-depth RNA feature directions as the reference, then projects each globally thinned RNA matrix onto those directions. For sampling rate rho, every RNA count is sampled with probability rho.

sampling_rates <- c(0.10, 0.20, 0.35, 0.50, 0.70, 0.85, 1.00)

umi_scaling <- scscale_umi_scaling(
  counts,
  sampling_rates = sampling_rates,
  r = 12,
  seed = 164378,
  n_replicates = 1,
  empirical = FALSE,
  mp_max_iter = 20,
  mp_grid_n = 2500
)

plot_total <- subset(umi_scaling$q_total, sampling_rate > 0.1)
plot_total_model <- lm(q_total ~ sampling_rate, data = plot_total)
plot_total$q_total_rate_hat <- predict(plot_total_model, newdata = plot_total)
plot_total_r2 <- summary(plot_total_model)$r.squared

data.frame(

Linear Fits

top_ranks <- 1:6
plot_q <- subset(umi_scaling$scaling, rank %in% top_ranks)
cols <- c("#0072B2", "#D55E00", "#009E73", "#CC79A7", "#56B4E9", "#E69F00")

old_par <- par(mfrow = c(1, 3), mar = c(4.2, 4.2, 2.4, 1))
plot(
  NA,
  xlim = range(plot_q$sampling_rate),
  ylim = range(c(plot_q$q_X, plot_q$q_X_rate_hat)),
  xlab = expression("Sampling rate " * rho),
  ylab = expression(q[X*k](rho)),
  main = "Top spike fits"
)
for (i in seq_along(top_ranks)) {
  k <- top_ranks[i]
  df <- plot_q[plot_q$rank == k, ]
  points(df$sampling_rate, df$q_X, pch = 16, col = cols[i])
  fit_df <- unique(df[, c("sampling_rate", "q_X_rate_hat")])
  fit_df <- fit_df[order(fit_df$sampling_rate), ]
  lines(fit_df$sampling_rate, fit_df$q_X_rate_hat, col = cols[i], lwd = 2)
}
legend(
  "topleft",
  legend = paste0("k=", top_ranks),
  col = cols,
  pch = 16,
  lwd = 2,
  bty = "n",
  cex = 0.85
)
